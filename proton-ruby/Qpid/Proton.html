def self.amqp_uri(s)
  u = URI(s)
  u.host ||= ""               # Behaves badly with nil host
  return u if u.is_a? URI::AMQP
  raise URI::BadURIError, "Not an AMQP URI: '#{u}'" if u.scheme
  u.scheme = "amqp" unless u.scheme
  u = URI::parse(u.to_s)      # Re-parse with amqp scheme
  return u
end

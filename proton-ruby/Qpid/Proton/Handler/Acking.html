def release(delivery, delivered = true)
  if delivered
    self.settle(delivery, Qpid::Proton::Delivery::MODIFIED)
  else
    self.settle(delivery, Qpid::Proton::Delivery::RELEASED)
  end
end

def settle(delivery, state = nil)
